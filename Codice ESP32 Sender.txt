void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);  // ESP32 come stazione per mandare dati

  // Inizializzare ESP-NOW
  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  esp_now_peer_info_t peerInfo;
  memcpy(peerInfo.peer_addr, RECEIVER_MAC_ADDR, 6);  
  peerInfo.channel = 0;                              
  peerInfo.encrypt = false;

  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("ESPNOW fallito");
    return;
  }

  Serial.println("ESP-NOW inizializzato con successo!");

  screen.begin( SSD1306_SWITCHCAPVCC, SCREEN_I2C_ADDRESS);
  screen.clearDisplay();
  screen.setTextSize(1);
  screen.setTextColor(WHITE);
  screen.setCursor(0,0);
  Serial.println("Setup finito");
  screen.print("Setup fatto!!");
  screen.display(); 
delay(100);
}

void loop() {
  x = analogRead(34);
  Serial.println(x);
  y = analogRead(35);
  Serial.println(y);
  x = map(x, 0, 4096, 0, 1000);
  y = map(y, 0, 4096, 0, 1000);
  outgoingMessage.integerData = x;

  // Manda le variabile
  esp_err_t result = esp_now_send(RECEIVER_MAC_ADDR, (uint8_t *)&outgoingMessage, sizeof(outgoingMessage));
  outgoingMessage.integerData = y;
  esp_now_send(RECEIVER_MAC_ADDR, (uint8_t *)&outgoingMessage, sizeof(outgoingMessage));
  
  if (result == ESP_OK) {
    Serial.println("Messaggio mandato");
    Serial.print("Valore x :");
    Serial.println(x);
    Serial.print("Valore y :");
    Serial.println(y);
  } else {
    Serial.println("Error");
  }

  screen.clearDisplay();

 x = map(x, 0, 1000, 0, 180);
  y = map(y, 0, 1000, 0, 180);

if(y>95) screen.drawBitmap(0,0, epd_bitmap_I2C_men__su , 128, 64, WHITE);
if(y<75) screen.drawBitmap(0,0, epd_bitmap_I2C_men__gi_, 128, 64, WHITE);  
if(x>95) screen.drawBitmap(0,0, epd_bitmap_I2C_men__destra, 128, 64, WHITE);
if(x<75) screen.drawBitmap(0,0, epd_bitmap_I2C_men__sinistra, 128, 64, WHITE);       
if((x<95)&&(y<95)&&(x>75)&&(y>75)) screen.drawBitmap(0,0, epd_bitmap_I2C_men_ , 128, 64, WHITE); 
  screen.display(); 
  delay(200);  // Manda dati ogni 0,2 secondi

}
